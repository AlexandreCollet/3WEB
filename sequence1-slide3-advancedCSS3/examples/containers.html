Box Shadow

Position (h-shadow & v-shadow)

div { box-shadow: 5px 10px 0px 0px white; }
div { box-shadow: -10px -10px 0px 0px white; }

Blur

div { box-shadow: 10px 10px 5px 0px white; }
div { box-shadow: 0px 0px 25px 0px white; }

Spread

div { box-shadow: 0px 0px 5px 10px white; }

Inset

div { box-shadow: 0px 0px 25px 0px white inset; }

Box Sizing

div { box-sizing: padding-box|border-box|content-box; }

Content box (default)

div {
	box-sizing : content-box;
	width      : 200px;
	height     : 100px;
	padding    : 5px;
	border     : 10px solid white;
}

Border box

div {
	box-sizing : border-box;
	width      : 200px;
	height     : 100px;
	padding    : 5px;
	border     : 10px solid white;
}

div { border-radius : 25px 0 0 0; }
